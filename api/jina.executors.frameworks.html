class jina.executors.frameworks.BaseTorchExecutor(model_name=None, *args, **kwargs)[source]¬∂

Bases: jina.executors.frameworks.BaseFrameworkExecutor

BaseTorchExecutor implements the base class for the executors using torch library. The common setups

go into this class.

To implement your own executor with the torch library,

class MyAwesomeTorchEncoder(BaseTorchExecutor):
    def post_init(self):
        # load your awesome model
        import torchvision.models as models
        self.model = models.mobilenet_v2().features.eval()
        self.to_device(self.model)

    def encode(self, data, *args, **kwargs):
        # use your awesome model to encode/craft/score
        import torch
        _input = torch.from_numpy(data)
        if self.on_gpu:
            _input = _input.cuda()
        _output = self.model(_input).detach()
        if self.on_gpu:
            _output = _output.cpu()
        return _output.numpy()
